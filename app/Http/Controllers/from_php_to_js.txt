<?php

$data = []; // в этот массив запишем то, что выберем из базы

$result = mysqli_query("SELECT * FROM pedestrianWays"); // сделаем запрос в БД
while($row = mysqli_fetch_assoc($result)){ // оформим каждую строку результата
    // как ассоциативный массив
    $data[] = $row; // допишем строку из выборки как новый элемент результирующего массива
}
echo json_encode($data); // и отдаём как json
